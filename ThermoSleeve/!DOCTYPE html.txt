        // สถานะสินค้าที่ถูกเลือกขณะนั้น (ชิ้นเริ่มต้นเป็นรุ่นหัวเข่า)
        let selectedProductId = 2;
        let itemQuantity = 1;
        let promoCodeDiscount = 0;

        // ฟังก์ชันเมื่อเปลี่ยนชิ้นสินค้า
        function selectProduct(id) {
            selectedProductId = id;
            itemQuantity = 1; // รีเซ็ตจำนวน
            syncProductDataToUI();
            changePage('product-page');
        }

        function openDefaultProduct() {
            syncProductDataToUI();
            changePage('product-page');
        }

        // ฟังก์ชันเชื่อมข้อมูลจาก Database ไปแสดงผลที่ UI แต่ละส่วน
        function syncProductDataToUI() {
            const data = productsDatabase[selectedProductId];
            
            // หน้าผลิตภัณฑ์รายละเอียด (Product Page)
            document.getElementById('main-product-viewer').src = data.image;
            document.getElementById('thumb-1').src = data.image;
            document.getElementById('dyn-product-title').innerText = data.title;
            document.getElementById('dyn-product-price').innerText = `$${data.price.toFixed(2)}`;
            document.getElementById('dyn-product-desc').innerText = data.desc;

            // หน้าตะกร้าสินค้า (Shopping Cart)
            document.getElementById('cart-img').src = data.image;
            document.getElementById('cart-title').innerText = data.title;
            document.getElementById('cart-single-price').innerText = `$${data.price.toFixed(2)}`;

            // หน้าชำระเงิน (Checkout)
            document.getElementById('chk-img').src = data.image;
            document.getElementById('chk-title').innerText = data.title;

            refreshCartSummaryCalculator();
        }

        // ระบบสลับหน้าจอ (SPA Simulation Routing)
        function changePage(pageId) {
            document.querySelectorAll('.page-view').forEach(view => {
                view.classList.add('hidden');
            });
            document.getElementById(pageId).classList.remove('hidden');
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        // ระบบเลื่อนลงไปยังสัดส่วนหน้าแรก
        function scrollToSection(sectionId) {
            changePage('homepage');
            setTimeout(() => {
                const target = document.getElementById(sectionId);
                if(target) target.scrollIntoView({ behavior: 'smooth', block: 'start' });
            }, 100);
        }

        function toggleMobileMenu() {
            const menu = document.getElementById('mobile-menu');
            menu.classList.toggle('hidden');
        }

        // คำนวณขนาดสัดส่วน (Interactive Size Advisor)
        function calculateProductSize() {
            const inputVal = document.getElementById('size-calc-input').value;
            const resultBox = document.getElementById('size-calc-result');
            
            if(!inputVal || inputVal <= 0) {
                resultBox.innerText = "❌ โปรดป้อนขนาดเส้นรอบวงส่วนที่กระชับรัดให้ถูกต้องด้วยครับ";
                resultBox.className = "text-xs font-bold text-red-600 bg-red-50 p-2 rounded-lg mt-3 animate-pulse";
                resultBox.classList.remove('hidden');
                return;
            }

            let sizeLabel = "XL";
            if(inputVal >= 30 && inputVal <= 33) sizeLabel = "S";
            else if(inputVal >= 34 && inputVal <= 37) sizeLabel = "M";
            else if(inputVal >= 38 && inputVal <= 41) sizeLabel = "L";

            resultBox.innerHTML = `✨ ไซส์ผลิตภัณฑ์ที่แนะนำสำหรับสัดส่วนคุณคือขนาด: <span class="bg-slate-900 text-white px-2.5 py-0.5 rounded ml-1 font-bold">Size ${sizeLabel}</span> (กระชับพอดีสรีระระบายเหงื่อคล่องตัว)`;
            resultBox.className = "text-xs font-bold text-slate-900 bg-neonGreen/40 p-2 rounded-lg mt-3 shadow-sm";
            resultBox.classList.remove('hidden');
        }

        // คำนวณสรุปราคาสุทธิ
        function refreshCartSummaryCalculator() {
            const data = productsDatabase[selectedProductId];
            let subTotal = data.price * itemQuantity;
            let finalNetPay = subTotal - promoCodeDiscount;

            document.getElementById('cart-quantity-val').innerText = itemQuantity;
            document.getElementById('subtotal-val').innerText = `$${subTotal.toFixed(2)}`;
            document.getElementById('discount-val').innerText = `-$${promoCodeDiscount.toFixed(2)}`;
            document.getElementById('total-val').innerText = `$${finalNetPay.toFixed(2)}`;
            
            document.getElementById('checkout-final-qty').innerText = itemQuantity;
            document.getElementById('checkout-final-total').innerText = `$${finalNetPay.toFixed(2)}`;
            document.getElementById('checkout-subtotal-val').innerText = `$${finalNetPay.toFixed(2)}`;
            document.getElementById('cart-badge-count').innerText = itemQuantity;
        }

        function changeQty(amount) {
            itemQuantity += amount;
            if (itemQuantity < 1) itemQuantity = 1;
            refreshCartSummaryCalculator();
        }

        function simulateAddToCart() {
            alert("🛍️ สั่งการสำเร็จ! เพิ่มผลิตภัณฑ์ ThermoSleeve รุ่นที่คุณเลือกเข้าสู่ระบบตะกร้าเรียบร้อย");
            refreshCartSummaryCalculator();
        }

        function clearCart() {
            document.getElementById('cart-items-wrapper').innerHTML = `<p class="text-slate-400 text-sm text-center py-6">ไม่มีรายการสินค้าคงเหลือในตะกร้าคำสั่งซื้อของคุณขณะนี้</p>`;
            document.getElementById('subtotal-val').innerText = "$0.00";
            document.getElementById('total-val').innerText = "$0.00";
            document.getElementById('cart-badge-count').innerText = "0";
            document.getElementById('checkout-item-preview').innerHTML = `<p class="text-slate-400 text-xs text-center py-2">ไม่มีสินค้าที่ค้างชำระเงิน</p>`;
            document.getElementById('checkout-subtotal-val').innerText = "$0.00";
        }

        function applyCartCoupon() {
            const inputField = document.getElementById('coupon-code-input').value.trim().toUpperCase();
            const msgLabel = document.getElementById('coupon-status-msg');
            
            if(inputField === "EARLYBIRD") {
                promoCodeDiscount = 10.00; // ลด 10 ดอลลาร์ทันที
                msgLabel.innerText = "✅ เปิดใช้งานโค้ดแคมเปญ EARLYBIRD สำเร็จ รับส่วนลดเพิ่มพิเศษ -$10.00 ทันที";
                msgLabel.className = "text-xs font-semibold mt-1.5 text-emerald-700 bg-emerald-50 p-1.5 rounded";
                msgLabel.classList.remove('hidden');
            } else {
                promoCodeDiscount = 0;
                msgLabel.innerText = "❌ รหัสคูปองไม่ถูกต้องสำหรับสิทธิ์แคมเปญสั่งซื้อล่วงหน้านี้";
                msgLabel.className = "text-xs font-semibold mt-1.5 text-red-600 bg-red-50 p-1.5 rounded";
                msgLabel.classList.remove('hidden');
            }
            refreshCartSummaryCalculator();
        }

        // รันข้อมูลเริ่มต้นเมื่อเปิดหน้าเว็บครั้งแรก
        syncProductDataToUI();
    </script>
